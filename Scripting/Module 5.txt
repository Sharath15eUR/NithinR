Recursive Functions
    1)  vim recursion.sh
            #!/bin/bash
            calculate_sum() {
            local n=$1
            if [ $n -eq 1 ]; then
                echo 1
            else
                local prev_sum=$(calculate_sum $((n - 1)))
                echo $((n + prev_sum))
            fi
            }
            read -p "Enter the value of n: " N
            result=$(calculate_sum $N)
            echo "The sum of the first $N numbers is: $result"

basics of redirection
    1)  vim addition.c
            #include <stdio.h>
            int main() {
                int num1, num2;
                printf("Enter two numbers: ");
                scanf("%d %d", &num1, &num2);
                int sum = num1 + num2;
                printf("Sum: %d\n", sum);
                return 0;
            }

        vim redirection.sh
            #!/bin/bash
            gcc -o addition addition.c > output.txt 2>&1
            if [ $? -eq 0 ]; then
                echo "Program compiled successfully."
            else
                echo "Compilation failed. Check output.txt for details."
            fi
